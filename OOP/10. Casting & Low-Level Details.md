## Cast Operators in C++
C++ provides four explicit cast operators for type conversions, each with specific purposes and safety guarantees.

**Overview**

```
┌─────────────────────────────────────────────────────────────────┐
│                    C++ CAST OPERATORS                           │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  static_cast<T>        ✓ Compile-time checked                  │
│                        ✓ Well-defined conversions               │
│                        ✓ No runtime overhead                    │
│                        • Upcasting/downcasting                  │
│                        • Numeric conversions                    │
│                        • void* conversions                      │
│                                                                 │
│  dynamic_cast<T>       ✓ Runtime type checking (RTTI)          │
│                        ✓ Safe downcasting                       │
│                        ✗ Requires polymorphic types             │
│                        ✗ Runtime overhead                       │
│                        • Polymorphic hierarchies                │
│                        • Returns nullptr on failure             │
│                                                                 │
│  const_cast<T>         ✓ Add/remove const/volatile             │
│                        ✗ Can break const-correctness            │
│                        • Interfacing with legacy code           │
│                        • Rare legitimate uses                   │
│                                                                 │
│  reinterpret_cast<T>   ✗ Dangerous bit-level reinterpretation  │
│                        ✗ Implementation-defined                 │
│                        ✗ No safety checks                       │
│                        • Low-level programming                  │
│                        • Hardware interaction                   │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

### 1. static_cast - Compile-Time Conversions
**Purpose:** Well-defined, compile-time checked conversions between related types.

**Basic Type Conversions:**

``` cpp
#include <iostream>
using namespace std;

int main() {
    // ═══════════════════════════════════════════════════════
    // Numeric conversions
    // ═══════════════════════════════════════════════════════
    double pi = 3.14159;
    int intPi = static_cast<int>(pi);  // 3 (truncation)
    cout << "double to int: " << intPi << endl;
    
    int num = 65;
    char ch = static_cast<char>(num);  // 'A'
    cout << "int to char: " << ch << endl;
    
    // ═══════════════════════════════════════════════════════
    // Explicit conversions (prevents narrowing warnings)
    // ═══════════════════════════════════════════════════════
    float f = 3.14f;
    int i = static_cast<int>(f);  // Clear intent
    // int i = f;  // ⚠️ Implicit conversion - may warn
    
    // ═══════════════════════════════════════════════════════
    // Enum conversions
    // ═══════════════════════════════════════════════════════
    enum Color { RED, GREEN, BLUE };
    int colorValue = static_cast<int>(GREEN);  // 1
    Color color = static_cast<Color>(2);       // BLUE
    
    cout << "Enum value: " << colorValue << endl;
    
    // ═══════════════════════════════════════════════════════
    // void* conversions
    // ═══════════════════════════════════════════════════════
    int value = 42;
    void* voidPtr = &value;
    
    // Must use static_cast to convert back
    int* intPtr = static_cast<int*>(voidPtr);
    cout << "Value from void*: " << *intPtr << endl;
    
    return 0;
}
```

**Output:**

```text

double to int: 3
int to char: A
Enum value: 1
Value from void*: 42
```

**Upcasting and Downcasting in Inheritance:**
``` cpp
#include <iostream>
using namespace std;

class Animal {
public:
    virtual void makeSound() { cout << "Animal sound" << endl; }
    virtual ~Animal() = default;
};

class Dog : public Animal {
public:
    void makeSound() override { cout << "Woof!" << endl; }
    void fetch() { cout << "Fetching ball" << endl; }
};

class Cat : public Animal {
public:
    void makeSound() override { cout << "Meow!" << endl; }
    void climb() { cout << "Climbing tree" << endl; }
};

int main() {
    // ═══════════════════════════════════════════════════════
    // UPCASTING: Derived → Base (always safe)
    // ═══════════════════════════════════════════════════════
    Dog dog;
    Animal* animalPtr = static_cast<Animal*>(&dog);  // Safe
    // Actually, upcasting is implicit - cast not needed
    Animal* animalPtr2 = &dog;  // Preferred way
    
    animalPtr->makeSound();  // Calls Dog::makeSound()
    
    // ═══════════════════════════════════════════════════════
    // DOWNCASTING: Base → Derived (⚠️ DANGEROUS without checks)
    // ═══════════════════════════════════════════════════════
    Animal* basePtr = new Dog();
    
    // ⚠️ Programmer must ENSURE this is actually a Dog
    Dog* dogPtr = static_cast<Dog*>(basePtr);
    dogPtr->fetch();  // OK - it really is a Dog
    
    // ❌ DANGER: Wrong downcast - UNDEFINED BEHAVIOR
    Animal* catBasePtr = new Cat();
    Dog* wrongCast = static_cast<Dog*>(catBasePtr);
    // wrongCast->fetch();  // 💥 UNDEFINED BEHAVIOR!
    
    delete basePtr;
    delete catBasePtr;
    
    return 0;
}
```

**Converting Between Pointer Types:**
``` cpp
#include <iostream>
using namespace std;

class Base {
public:
    int baseData = 100;
    virtual ~Base() = default;
};

class Derived : public Base {
public:
    int derivedData = 200;
};

int main() {
    Derived d;
    
    // Pointer conversions
    Derived* dPtr = &d;
    
    // Upcast: Derived* → Base*
    Base* bPtr = static_cast<Base*>(dPtr);  // Safe
    cout << "Base data: " << bPtr->baseData << endl;
    
    // Downcast: Base* → Derived* (programmer responsibility)
    Derived* dPtr2 = static_cast<Derived*>(bPtr);
    cout << "Derived data: " << dPtr2->derivedData << endl;
    
    // ❌ ERROR: Cannot cast between unrelated types
    // class Unrelated {};
    // Unrelated* uPtr = static_cast<Unrelated*>(bPtr);  // Compile error
    
    return 0;
}
```

### 2. dynamic_cast - Safe Runtime Downcasting
**Purpose:** Safe downcasting in polymorphic hierarchies with runtime type checking.

**Requirements:**
✅ Class must have at least one virtual function (RTTI enabled)
✅ Returns nullptr for pointers on failure
✅ Throws std::bad_cast for references on failure

**Basic Usage:**
``` cpp
#include <iostream>
#include <typeinfo>
using namespace std;

class Shape {
public:
    virtual void draw() const { cout << "Drawing shape" << endl; }
    virtual ~Shape() = default;
};

class Circle : public Shape {
private:
    double radius;
public:
    Circle(double r) : radius(r) {}
    void draw() const override { 
        cout << "Drawing circle (radius=" << radius << ")" << endl; 
    }
    double getRadius() const { return radius; }
};

class Rectangle : public Shape {
private:
    double width, height;
public:
    Rectangle(double w, double h) : width(w), height(h) {}
    void draw() const override { 
        cout << "Drawing rectangle (" << width << "x" << height << ")" << endl; 
    }
    double getArea() const { return width * height; }
};

int main() {
    Shape* shapes[3];
    shapes[0] = new Circle(5.0);
    shapes[1] = new Rectangle(10, 20);
    shapes[2] = new Circle(3.0);
    
    for (int i = 0; i < 3; i++) {
        cout << "\n=== Shape " << i << " ===" << endl;
        shapes[i]->draw();
        
        // Try to downcast to Circle
        Circle* circle = dynamic_cast<Circle*>(shapes[i]);
        if (circle) {
            cout << "✓ This is a Circle! Radius: " << circle->getRadius() << endl;
        }
        
        // Try to downcast to Rectangle
        Rectangle* rect = dynamic_cast<Rectangle*>(shapes[i]);
        if (rect) {
            cout << "✓ This is a Rectangle! Area: " << rect->getArea() << endl;
        }
    }
    
    // Cleanup
    for (int i = 0; i < 3; i++) {
        delete shapes[i];
    }
    
    return 0;
}
```

**Output:**

```text

=== Shape 0 ===
Drawing circle (radius=5)
✓ This is a Circle! Radius: 5

=== Shape 1 ===
Drawing rectangle (10x20)
✓ This is a Rectangle! Area: 200

=== Shape 2 ===
Drawing circle (radius=3)
✓ This is a Circle! Radius: 3
```

**Reference Casting (Throws on Failure):**

``` cpp
#include <iostream>
#include <exception>
using namespace std;

class Base {
public:
    virtual ~Base() = default;
};

class Derived : public Base {
public:
    void derivedMethod() { cout << "Derived method called" << endl; }
};

int main() {
    Base base;
    Derived derived;
    
    Base& baseRef = derived;
    
    try {
        // ✅ Success: baseRef actually refers to Derived
        Derived& derivedRef = dynamic_cast<Derived&>(baseRef);
        derivedRef.derivedMethod();
        
        // ❌ Failure: base is not a Derived
        Base& actualBase = base;
        Derived& wrongRef = dynamic_cast<Derived&>(actualBase);
        
    } catch (const bad_cast& e) {
        cout << "❌ dynamic_cast failed: " << e.what() << endl;
    }
    
    return 0;
}
```

**Output:**

```text

Derived method called
❌ dynamic_cast failed: std::bad_cast
```

**Cross-Casting in Multiple Inheritance:**
``` cpp
#include <iostream>
using namespace std;

class Base1 {
public:
    virtual ~Base1() = default;
    void base1Method() { cout << "Base1 method" << endl; }
};

class Base2 {
public:
    virtual ~Base2() = default;
    void base2Method() { cout << "Base2 method" << endl; }
};

class Derived : public Base1, public Base2 {
public:
    void derivedMethod() { cout << "Derived method" << endl; }
};

int main() {
    Derived d;
    
    // Get Base2 pointer
    Base2* b2Ptr = &d;
    
    // Cross-cast: Base2* → Base1* (through Derived)
    Base1* b1Ptr = dynamic_cast<Base1*>(b2Ptr);
    
    if (b1Ptr) {
        cout << "✓ Cross-cast successful!" << endl;
        b1Ptr->base1Method();
    }
    
    // Also works for downcasting
    Derived* dPtr = dynamic_cast<Derived*>(b2Ptr);
    if (dPtr) {
        dPtr->derivedMethod();
    }
    
    return 0;
}
```

**Performance Comparison:**
``` cpp
#include <iostream>
#include <chrono>
using namespace std;

class Base {
public:
    virtual ~Base() = default;
};

class Derived : public Base { };

int main() {
    const int ITERATIONS = 10000000;
    Base* ptr = new Derived();
    
    // ═══════════════════════════════════════════════════════
    // static_cast (compile-time, fast)
    // ═══════════════════════════════════════════════════════
    auto start = chrono::high_resolution_clock::now();
    for (int i = 0; i < ITERATIONS; i++) {
        Derived* d = static_cast<Derived*>(ptr);
        (void)d;  // Prevent optimization
    }
    auto end = chrono::high_resolution_clock::now();
    auto staticDuration = chrono::duration_cast<chrono::milliseconds>(end - start);
    
    // ═══════════════════════════════════════════════════════
    // dynamic_cast (runtime, slower)
    // ═══════════════════════════════════════════════════════
    start = chrono::high_resolution_clock::now();
    for (int i = 0; i < ITERATIONS; i++) {
        Derived* d = dynamic_cast<Derived*>(ptr);
        (void)d;
    }
    end = chrono::high_resolution_clock::now();
    auto dynamicDuration = chrono::duration_cast<chrono::milliseconds>(end - start);
    
    cout << "static_cast:  " << staticDuration.count() << " ms" << endl;
    cout << "dynamic_cast: " << dynamicDuration.count() << " ms" << endl;
    cout << "Overhead: " << (dynamicDuration.count() - staticDuration.count()) 
         << " ms (" << (100.0 * dynamicDuration.count() / staticDuration.count() - 100) 
         << "% slower)" << endl;
    
    delete ptr;
    return 0;
}
```

### 3. const_cast - Modifying Const-ness
Purpose: Add or remove const and volatile qualifiers.

⚠️ **Warning:** Using const_cast to modify a truly const object is undefined behavior!

**Removing const (Dangerous):**

``` cpp
#include <iostream>
using namespace std;

void modifyValue(int* ptr) {
    *ptr = 100;
}

int main() {
    // ═══════════════════════════════════════════════════════
    // ✅ SAFE: Original object is non-const
    // ═══════════════════════════════════════════════════════
    int value = 42;
    const int* constPtr = &value;  // const pointer to non-const data
    
    cout << "Before: " << value << endl;
    
    // Remove const to pass to legacy function
    int* mutablePtr = const_cast<int*>(constPtr);
    modifyValue(mutablePtr);
    
    cout << "After: " << value << endl;  // 100 - OK!
    
    // ═══════════════════════════════════════════════════════
    // ❌ UNDEFINED BEHAVIOR: Original object is const
    // ═══════════════════════════════════════════════════════
    const int constValue = 50;
    const int* constPtr2 = &constValue;
    
    int* badPtr = const_cast<int*>(constPtr2);
    // *badPtr = 200;  // 💥 UNDEFINED BEHAVIOR - don't do this!
    
    return 0;
}
```

**Legitimate Use Case: Legacy API**

``` cpp
#include <iostream>
#include <cstring>
using namespace std;

// Legacy C-style function (doesn't modify but lacks const)
void legacyPrint(char* str) {
    cout << "Legacy print: " << str << endl;
}

// Modern C++ function with const-correctness
void modernFunction(const char* message) {
    // Need to call legacy function
    // const_cast is necessary to interface with old code
    legacyPrint(const_cast<char*>(message));
}

int main() {
    const char* greeting = "Hello, World!";
    modernFunction(greeting);
    return 0;
}
```

**Removing const in Class Methods:**
``` cpp
#include <iostream>
using namespace std;

class Cache {
private:
    mutable int cachedValue;  // Better alternative to const_cast
    mutable bool cacheValid;
    
    int expensiveComputation() const {
        cout << "Computing..." << endl;
        return 42;
    }

public:
    Cache() : cachedValue(0), cacheValid(false) {}
    
    // const method that needs to modify cache
    int getValue() const {
        if (!cacheValid) {
            // Method 1: Using mutable (preferred)
            cachedValue = expensiveComputation();
            cacheValid = true;
        }
        return cachedValue;
    }
    
    // Method 2: Using const_cast (not recommended)
    int getValueBad() const {
        if (!cacheValid) {
            // ⚠️ Casting away const from 'this'
            Cache* mutableThis = const_cast<Cache*>(this);
            mutableThis->cachedValue = expensiveComputation();
            mutableThis->cacheValid = true;
        }
        return cachedValue;
    }
};

int main() {
    const Cache cache;
    cout << cache.getValue() << endl;  // Computes
    cout << cache.getValue() << endl;  // Uses cache
    return 0;
}
```

### 4. reinterpret_cast - Low-Level Bit Reinterpretation
**Purpose:** Platform-dependent, low-level casts. Reinterprets bit pattern.

⚠️ **Warning:** Extremely dangerous! Use only when absolutely necessary.

**Pointer Type Reinterpretation:**

``` cpp
#include <iostream>
using namespace std;

int main() {
    // ═══════════════════════════════════════════════════════
    // Converting pointer to integer
    // ═══════════════════════════════════════════════════════
    int value = 42;
    int* ptr = &value;
    
    // Store pointer as integer (for hashing, debugging, etc.)
    uintptr_t address = reinterpret_cast<uintptr_t>(ptr);
    cout << "Pointer address: 0x" << hex << address << dec << endl;
    
    // Convert back
    int* restoredPtr = reinterpret_cast<int*>(address);
    cout << "Value: " << *restoredPtr << endl;
    
    // ═══════════════════════════════════════════════════════
    // Examining object representation
    // ═══════════════════════════════════════════════════════
    float f = 3.14f;
    
    // View float's bit representation as unsigned int
    unsigned int* bits = reinterpret_cast<unsigned int*>(&f);
    cout << "Float bits: 0x" << hex << *bits << dec << endl;
    
    return 0;
}
```
**Function Pointer Casting:**

``` cpp
#include <iostream>
using namespace std;

void myFunction() {
    cout << "Hello from function!" << endl;
}

typedef void (*FuncPtr)();

int main() {
    // Get function address
    void* voidPtr = reinterpret_cast<void*>(&myFunction);
    
    // Convert back to function pointer
    FuncPtr funcPtr = reinterpret_cast<FuncPtr>(voidPtr);
    
    funcPtr();  // Call the function
    
    return 0;
}
```

**Type Punning (Dangerous!):**
``` cpp
#include <iostream>
#include <cstring>
using namespace std;

int main() {
    float f = 3.14159f;
    
    // ❌ WRONG: Undefined behavior
    // int i = *reinterpret_cast<int*>(&f);

    cout << "As int: " << i << " (0x" << hex << i << dec << ")" << endl;
    
    // C++20 alternative: std::bit_cast
    // int i = std::bit_cast<int>(f);
    
    return 0;
}
```

### 5. Comparison and When to Use
``` cpp
#include <iostream>
using namespace std;

class Base {
public:
    virtual ~Base() = default;
    int baseData = 100;
};

class Derived : public Base {
public:
    int derivedData = 200;
};

void demonstrateCasts() {
    Derived d;
    Base* basePtr = &d;
    
    cout << "=== static_cast ===" << endl;
    // Compile-time check, no runtime overhead
    Derived* d1 = static_cast<Derived*>(basePtr);
    cout << "Derived data: " << d1->derivedData << endl;
    
    cout << "\n=== dynamic_cast ===" << endl;
    // Runtime check, safe but slower
    Derived* d2 = dynamic_cast<Derived*>(basePtr);
    if (d2) {
        cout << "Safe downcast succeeded" << endl;
        cout << "Derived data: " << d2->derivedData << endl;
    }
    
    cout << "\n=== const_cast ===" << endl;
    const Base* constPtr = basePtr;
    Base* mutablePtr = const_cast<Base*>(constPtr);
    mutablePtr->baseData = 999;
    cout << "Modified base data: " << d.baseData << endl;
    
    cout << "\n=== reinterpret_cast ===" << endl;
    // Low-level, dangerous
    uintptr_t address = reinterpret_cast<uintptr_t>(basePtr);
    cout << "Pointer as integer: 0x" << hex << address << dec << endl;
}

int main() {
    demonstrateCasts();
    return 0;
}
```

### 7. Common Mistakes

``` cpp
#include <iostream>
using namespace std;

class Base {
public:
    virtual ~Base() = default;
};

class Derived : public Base { };

// ❌ MISTAKE 1: Using static_cast when dynamic_cast is needed
void mistake1() {
    Base* basePtr = new Base();  // Not actually a Derived!
    
    // ⚠️ No safety check - undefined behavior!
    Derived* wrongCast = static_cast<Derived*>(basePtr);
    // Using wrongCast is undefined behavior
    
    // ✅ CORRECT: Use dynamic_cast
    Derived* safeCast = dynamic_cast<Derived*>(basePtr);
    if (!safeCast) {
        cout << "Not a Derived object" << endl;
    }
    
    delete basePtr;
}

// ❌ MISTAKE 2: Modifying truly const objects
void mistake2() {
    const int constValue = 42;
    
    // ⚠️ Undefined behavior!
    // int* ptr = const_cast<int*>(&constValue);
    // *ptr = 100;  // 💥 May crash or produce weird results
}

// ❌ MISTAKE 3: reinterpret_cast for type punning
void mistake3() {
    float f = 3.14f;
    
    // ⚠️ Undefined behavior (strict aliasing violation)
    // int i = *reinterpret_cast<int*>(&f);
    
    // ✅ CORRECT: Use memcpy
    int i;
    memcpy(&i, &f, sizeof(f));
}

// ❌ MISTAKE 4: dynamic_cast on non-polymorphic type
// class NonPolymorphic { };  // No virtual functions
// void mistake4() {
//     NonPolymorphic np;
//     // Compile error: Not a polymorphic type
//     // void* ptr = dynamic_cast<void*>(&np);
// }

int main() {
    mistake1();
    mistake2();
    mistake3();
    return 0;
}
```

### 8. Best Practices Summary
``` cpp
// ═══════════════════════════════════════════════════════════
// BEST PRACTICES
// ═══════════════════════════════════════════════════════════

// ✅ DO: Prefer static_cast for well-known conversions
double d = 3.14;
int i = static_cast<int>(d);

// ✅ DO: Use dynamic_cast for safe polymorphic downcasts
Base* base = getObject();
if (Derived* derived = dynamic_cast<Derived*>(base)) {
    // Safe to use derived
}

// ✅ DO: Use const_cast only when interfacing with legacy code
void legacyFunction(char* str);  // Can't change signature
void modernWrapper(const char* str) {
    legacyFunction(const_cast<char*>(str));
}

// ✅ DO: Avoid reinterpret_cast unless doing low-level work
// Only in embedded systems, hardware access, etc.

// ❌ DON'T: Use C-style casts
// int i = (int)d;  // Which cast is this? Unclear and dangerous!

// ❌ DON'T: Cast away const to modify
// const int* ptr = &value;
// *const_cast<int*>(ptr) = 10;  // Dangerous!

// ❌ DON'T: Use reinterpret_cast for type punning
// Use memcpy or std::bit_cast instead
```

### Key Takeaways:
**static_cast:** Default choice for well-defined conversions
**dynamic_cast:** Use for safe polymorphic downcasting
**const_cast:** Rarely needed, mainly for legacy code
**reinterpret_cast:** Expert-level only, extremely dangerous

**Golden Rule:** If you're not sure which cast to use, you probably shouldn't be casting at all! Reconsider your design. 🎯

## Nested/Inner Classes in C++
Nested classes are classes defined inside another class. They provide encapsulation, logical grouping, and implementation hiding.

**Overview:**
```
┌─────────────────────────────────────────────────────────────────┐
│                    NESTED CLASSES                               │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  class Outer {                                                  │
│  public:                                                        │
│      class Inner {        // Nested class                       │
│          // Inner class members                                 │
│      };                                                         │
│  private:                                                       │
│      class PrivateInner { // Private nested class               │
│          // Implementation detail                               │
│      };                                                         │
│  };                                                             │
│                                                                 │
│  KEY POINTS:                                                    │
│  • Nested class can access outer class's static members        │
│  • Nested class CANNOT access outer instance members directly  │
│  • Outer class CANNOT access nested class's private members    │
│  • Provides strong encapsulation and scope management          │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

### 1. Basic Nested Class
**Simple Example:**
``` cpp
#include <iostream>
#include <string>
using namespace std;

class Outer {
private:
    int outerData;
    static int staticData;

public:
    Outer(int data) : outerData(data) {
        cout << "Outer constructed with " << data << endl;
    }

    // ═══════════════════════════════════════════════════════
    // Public nested class
    // ═══════════════════════════════════════════════════════
    class Inner {
    private:
        int innerData;

    public:
        Inner(int data) : innerData(data) {
            cout << "Inner constructed with " << data << endl;
        }

        void display() const {
            cout << "Inner data: " << innerData << endl;
            
            // ✅ Can access outer's static members
            cout << "Outer static data: " << staticData << endl;
            
            // ❌ Cannot access outer's instance members directly
            // cout << outerData << endl;  // ERROR!
        }

        void displayWithOuter(const Outer& outer) const {
            cout << "Inner data: " << innerData << endl;
            cout << "Outer data (via reference): " << outer.outerData << endl;
        }
    };

    void createInner() {
        Inner inner(100);
        inner.display();
    }

    void showInnerData(const Inner& inner) {
        // Outer class can access public members of Inner
        inner.display();
        
        // ❌ Cannot access private members of Inner
        // cout << inner.innerData << endl;  // ERROR!
    }
};

// Initialize static member
int Outer::staticData = 42;

int main() {
    cout << "=== Creating Outer ===" << endl;
    Outer outer(10);

    cout << "\n=== Creating Inner (via Outer::Inner) ===" << endl;
    Outer::Inner inner1(20);  // Access via Outer::
    inner1.display();

    cout << "\n=== Inner accessing Outer via reference ===" << endl;
    inner1.displayWithOuter(outer);

    cout << "\n=== Outer creating Inner ===" << endl;
    outer.createInner();

    return 0;
}
```

### 2. Access Control
Public, Protected, and Private Nested Classes

``` cpp
#include <iostream>
using namespace std;

class Container {
private:
    int privateData = 100;

    // ═══════════════════════════════════════════════════════
    // Private nested class - implementation detail
    // ═══════════════════════════════════════════════════════
    class Node {
    private:
        int data;
        Node* next;

    public:
        Node(int d) : data(d), next(nullptr) {
            cout << "Node created: " << data << endl;
        }

        void setNext(Node* n) { next = n; }
        Node* getNext() const { return next; }
        int getData() const { return data; }

        // Node can access Container's private members if given reference
        void accessContainer(Container& c) {
            cout << "Container's private data: " << c.privateData << endl;
        }
    };

    Node* head;

public:
    Container() : head(nullptr) {
        cout << "Container created" << endl;
    }

    ~Container() {
        Node* current = head;
        while (current) {
            Node* next = current->getNext();
            delete current;
            current = next;
        }
    }

    void addNode(int data) {
        Node* newNode = new Node(data);
        newNode->setNext(head);
        head = newNode;
    }

    void display() const {
        cout << "Container contents: ";
        Node* current = head;
        while (current) {
            cout << current->getData() << " ";
            current = current->getNext();
        }
        cout << endl;
    }

    // ═══════════════════════════════════════════════════════
    // Public nested class - part of public interface
    // ═══════════════════════════════════════════════════════
    class Iterator {
    private:
        Node* current;

    public:
        Iterator(Node* node) : current(node) {}

        bool hasNext() const {
            return current != nullptr;
        }

        int next() {
            int data = current->getData();
            current = current->getNext();
            return data;
        }
    };

    Iterator begin() {
        return Iterator(head);
    }
};

int main() {
    Container container;
    container.addNode(10);
    container.addNode(20);
    container.addNode(30);

    container.display();

    cout << "\nIterating with public Iterator:" << endl;
    Container::Iterator it = container.begin();
    while (it.hasNext()) {
        cout << it.next() << " ";
    }
    cout << endl;

    // ❌ Cannot create private Node from outside
    // Container::Node node(5);  // ERROR: Node is private

    return 0;
}
```

**Output:**

```text

Container created
Node created: 10
Node created: 20
Node created: 30
Container contents: 30 20 10 

Iterating with public Iterator:
30 20 10
```

### 3. Nested Class with Outer Instance Access
Providing Outer Reference to Nested Class

``` cpp
#include <iostream>
#include <string>
using namespace std;

class BankAccount {
private:
    string accountNumber;
    double balance;
    
    static double interestRate;

public:
    BankAccount(string accNum, double initialBalance) 
        : accountNumber(accNum), balance(initialBalance) {
        cout << "Account " << accountNumber << " created with $" 
             << balance << endl;
    }

    // Nested class that needs access to outer instance
    // ═══════════════════════════════════════════════════════
    class Transaction {
    private:
        BankAccount& account;  // Reference to outer instance
        string type;
        double amount;

    public:
        Transaction(BankAccount& acc, string t, double amt)
            : account(acc), type(t), amount(amt) {}

        bool execute() {
            if (type == "deposit") {
                account.balance += amount;  // Access outer's private member!
                cout << "Deposited $" << amount << endl;
                return true;
            } 
            else if (type == "withdraw") {
                if (account.balance >= amount) {
                    account.balance -= amount;
                    cout << "Withdrew $" << amount << endl;
                    return true;
                } else {
                    cout << "Insufficient funds!" << endl;
                    return false;
                }
            }
            return false;
        }

        void displayAccountInfo() const {
            cout << "Account: " << account.accountNumber 
                 << ", Balance: $" << account.balance << endl;
        }

        static void applyInterest(BankAccount& acc) {
            acc.balance += acc.balance * interestRate;
            cout << "Interest applied. New balance: $" 
                 << acc.balance << endl;
        }
    };

    void performTransaction(string type, double amount) {
        Transaction trans(*this, type, amount);  // Pass *this
        trans.execute();
    }

    void displayBalance() const {
        cout << "Account " << accountNumber << ": $" << balance << endl;
    }
};

double BankAccount::interestRate = 0.05;

int main() {
    BankAccount account("ACC123", 1000.0);

    cout << "\n=== Direct transaction ===" << endl;
    BankAccount::Transaction trans1(account, "deposit", 500);
    trans1.execute();
    trans1.displayAccountInfo();

    cout << "\n=== Transaction via outer method ===" << endl;
    account.performTransaction("withdraw", 200);
    account.displayBalance();

    cout << "\n=== Applying interest ===" << endl;
    BankAccount::Transaction::applyInterest(account);
    account.displayBalance();

    return 0;
}
```

### 4. Nested Class in Templates

``` cpp
#include <iostream>
#include <stdexcept>
using namespace std;

template<typename T, size_t N>
class Array {
private:
    T data[N];

public:
    Array() {
        for (size_t i = 0; i < N; i++) {
            data[i] = T();
        }
    }

    T& operator[](size_t index) {
        if (index >= N) throw out_of_range("Index out of bounds");
        return data[index];
    }

    const T& operator[](size_t index) const {
        if (index >= N) throw out_of_range("Index out of bounds");
        return data[index];
    }

    size_t size() const { return N; }

    // ═══════════════════════════════════════════════════════
    // Nested Iterator class template
    // ═══════════════════════════════════════════════════════
    class Iterator {
    private:
        T* ptr;

    public:
        Iterator(T* p) : ptr(p) {}

        T& operator*() { return *ptr; }
        const T& operator*() const { return *ptr; }

        Iterator& operator++() {
            ++ptr;
            return *this;
        }

        Iterator operator++(int) {
            Iterator temp = *this;
            ++ptr;
            return temp;
        }

        bool operator!=(const Iterator& other) const {
            return ptr != other.ptr;
        }

        bool operator==(const Iterator& other) const {
            return ptr == other.ptr;
        }
    };

    Iterator begin() { return Iterator(data); }
    Iterator end() { return Iterator(data + N); }

    // ═══════════════════════════════════════════════════════
    // Nested ReverseIterator
    // ═══════════════════════════════════════════════════════
    class ReverseIterator {
    private:
        T* ptr;

    public:
        ReverseIterator(T* p) : ptr(p) {}

        T& operator*() { return *ptr; }

        ReverseIterator& operator++() {
            --ptr;
            return *this;
        }

        bool operator!=(const ReverseIterator& other) const {
            return ptr != other.ptr;
        }
    };

    ReverseIterator rbegin() { return ReverseIterator(data + N - 1); }
    ReverseIterator rend() { return ReverseIterator(data - 1); }
};

int main() {
    Array<int, 5> arr;

    // Fill array
    for (size_t i = 0; i < arr.size(); i++) {
        arr[i] = (i + 1) * 10;
    }

    cout << "=== Forward iteration ===" << endl;
    for (Array<int, 5>::Iterator it = arr.begin(); it != arr.end(); ++it) {
        cout << *it << " ";
    }
    cout << endl;

    cout << "=== Range-based for (uses Iterator) ===" << endl;
    for (int val : arr) {
        cout << val << " ";
    }
    cout << endl;

    cout << "=== Reverse iteration ===" << endl;
    for (Array<int, 5>::ReverseIterator it = arr.rbegin(); 
         it != arr.rend(); ++it) {
        cout << *it << " ";
    }
    cout << endl;

    return 0;
}
```

**Output:**

```text

=== Forward iteration ===
10 20 30 40 50 
=== Range-based for (uses Iterator) ===
10 20 30 40 50 
=== Reverse iteration ===
50 40 30 20 10
```

### 5. Real-World Example: Linked List
``` cpp
#include <iostream>
#include <stdexcept>
using namespace std;

template<typename T>
class LinkedList {
private:
    // ═══════════════════════════════════════════════════════
    // Private nested Node class - implementation detail
    // ═══════════════════════════════════════════════════════
    class Node {
    public:
        T data;
        Node* next;
        Node* prev;

        Node(const T& value) 
            : data(value), next(nullptr), prev(nullptr) {}
    };

    Node* head;
    Node* tail;
    size_t count;

public:
    LinkedList() : head(nullptr), tail(nullptr), count(0) {}

    ~LinkedList() {
        clear();
    }

    void pushBack(const T& value) {
        Node* newNode = new Node(value);
        
        if (!tail) {
            head = tail = newNode;
        } else {
            tail->next = newNode;
            newNode->prev = tail;
            tail = newNode;
        }
        ++count;
    }

    void pushFront(const T& value) {
        Node* newNode = new Node(value);
        
        if (!head) {
            head = tail = newNode;
        } else {
            newNode->next = head;
            head->prev = newNode;
            head = newNode;
        }
        ++count;
    }

    void popBack() {
        if (!tail) throw runtime_error("List is empty");
        
        Node* oldTail = tail;
        tail = tail->prev;
        
        if (tail) {
            tail->next = nullptr;
        } else {
            head = nullptr;
        }
        
        delete oldTail;
        --count;
    }

    void clear() {
        while (head) {
            Node* temp = head;
            head = head->next;
            delete temp;
        }
        tail = nullptr;
        count = 0;
    }

    size_t size() const { return count; }
    bool empty() const { return count == 0; }

    // ═══════════════════════════════════════════════════════
    // Public nested Iterator class
    // ═══════════════════════════════════════════════════════
    class Iterator {
    private:
        Node* current;
        friend class LinkedList;  // Allow LinkedList to access private members

    public:
        Iterator(Node* node) : current(node) {}

        T& operator*() {
            if (!current) throw runtime_error("Dereferencing null iterator");
            return current->data;
        }

        const T& operator*() const {
            if (!current) throw runtime_error("Dereferencing null iterator");
            return current->data;
        }

        Iterator& operator++() {
            if (current) current = current->next;
            return *this;
        }

        Iterator operator++(int) {
            Iterator temp = *this;
            if (current) current = current->next;
            return temp;
        }

        Iterator& operator--() {
            if (current) current = current->prev;
            return *this;
        }

        bool operator==(const Iterator& other) const {
            return current == other.current;
        }

        bool operator!=(const Iterator& other) const {
            return current != other.current;
        }
    };

    // ═══════════════════════════════════════════════════════
    // Public nested ConstIterator class
    // ═══════════════════════════════════════════════════════
    class ConstIterator {
    private:
        const Node* current;

    public:
        ConstIterator(const Node* node) : current(node) {}

        const T& operator*() const {
            if (!current) throw runtime_error("Dereferencing null iterator");
            return current->data;
        }

        ConstIterator& operator++() {
            if (current) current = current->next;
            return *this;
        }

        bool operator!=(const ConstIterator& other) const {
            return current != other.current;
        }
    };

    Iterator begin() { return Iterator(head); }
    Iterator end() { return Iterator(nullptr); }

    ConstIterator begin() const { return ConstIterator(head); }
    ConstIterator end() const { return ConstIterator(nullptr); }

    void display() const {
        cout << "List [" << count << " elements]: ";
        for (const auto& item : *this) {
            cout << item << " ";
        }
        cout << endl;
    }
};

int main() {
    LinkedList<int> list;

    cout << "=== Building list ===" << endl;
    list.pushBack(10);
    list.pushBack(20);
    list.pushBack(30);
    list.pushFront(5);
    list.display();

    cout << "\n=== Forward iteration ===" << endl;
    for (LinkedList<int>::Iterator it = list.begin(); it != list.end(); ++it) {
        cout << *it << " ";
    }
    cout << endl;

    cout << "\n=== Range-based for ===" << endl;
    for (int value : list) {
        cout << value << " ";
    }
    cout << endl;

    cout << "\n=== Modifying through iterator ===" << endl;
    for (auto it = list.begin(); it != list.end(); ++it) {
        *it *= 2;
    }
    list.display();

    cout << "\n=== Removing element ===" << endl;
    list.popBack();
    list.display();

    return 0;
}
```

**Output:**

```text

=== Building list ===
List [4 elements]: 5 10 20 30 

=== Forward iteration ===
5 10 20 30 

=== Range-based for ===
5 10 20 30 

=== Modifying through iterator ===
List [4 elements]: 10 20 40 60 

=== Removing element ===
List [3 elements]: 10 20 40

```

### 9. Comparison: Nested vs Regular Classes
``` cpp
#include <iostream>
using namespace std;

// ═══════════════════════════════════════════════════════
// Approach 1: Nested class
// ═══════════════════════════════════════════════════════
namespace NestedApproach {
    class Stack {
    private:
        class Node {  // Hidden implementation detail
        public:
            int data;
            Node* next;
            Node(int d) : data(d), next(nullptr) {}
        };

        Node* top;

    public:
        Stack() : top(nullptr) {}
        
        void push(int value) {
            Node* newNode = new Node(value);
            newNode->next = top;
            top = newNode;
        }

        // ... rest of implementation
    };

    // ✅ Advantages:
    // - Node is hidden from external code
    // - Clear that Node is part of Stack's implementation
    // - No namespace pollution
}

// ═══════════════════════════════════════════════════════
// Approach 2: Separate classes
// ═══════════════════════════════════════════════════════
namespace SeparateApproach {
    class StackNode {  // Exposed to everyone
    public:
        int data;
        StackNode* next;
        StackNode(int d) : data(d), next(nullptr) {}
    };

    class Stack {
    private:
        StackNode* top;

    public:
        Stack() : top(nullptr) {}
        
        void push(int value) {
            StackNode* newNode = new StackNode(value);
            newNode->next = top;
            top = newNode;
        }

        // ... rest of implementation
    };

    // ❌ Disadvantages:
    // - StackNode is visible to everyone
    // - Relationship between Stack and StackNode is unclear
    // - StackNode pollutes namespace
}

int main() {
    // Using nested approach
    NestedApproach::Stack stack1;
    stack1.push(10);
    
    // ❌ Cannot access nested Node from outside
    // NestedApproach::Stack::Node node(5);  // ERROR if private
    
    // Using separate approach
    SeparateApproach::Stack stack2;
    stack2.push(20);
    
    // ⚠️ Can create StackNode independently (may not be desired)
    SeparateApproach::StackNode node(5);
    
    return 0;
}
```

### Summary
**Feature	Description:**
Scope	Nested class name is scoped to outer class
Access	Can access outer's static members; needs reference for instance members
Visibility	Follows normal access control (public/private/protected)
Usage	Outer::Nested obj;
Common Use	Iterators, implementation details, helper classes

**Best Practices:**
✅ Use for tightly coupled helper classes
✅ Make private if implementation detail
✅ Make public if part of class interface (e.g., iterators)

❌ Don't overuse - can make code harder to navigate
❌ Avoid deep nesting - readability suffers

Nested classes are a powerful tool for encapsulation and logical organization! 🎯

### Enums in C++
Enumerations define named integral constants for better code readability and type safety. C++ has two types: plain enums and scoped enum classes.

**Overview:**
``` 
┌─────────────────────────────────────────────────────────────────┐
│                    ENUM TYPES IN C++                            │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  PLAIN ENUM (C-style)           SCOPED ENUM CLASS (C++11)      │
│  enum Color { ... };            enum class Color { ... };      │
│                                                                 │
│  • Unscoped names               • Scoped names                 │
│  • Implicit int conversion      • No implicit conversion       │
│  • Name pollution               • No name pollution            │
│  • Backward compatible          • Type safe                    │
```

### 4. Bitwise Operations (Flags)
``` cpp
#include <iostream>
using namespace std;

// ═══════════════════════════════════════════════════════
// Enum for bit flags
// ═══════════════════════════════════════════════════════
enum class Permissions : unsigned int {
    None    = 0,
    Read    = 1 << 0,  // 0001 = 1
    Write   = 1 << 1,  // 0010 = 2
    Execute = 1 << 2,  // 0100 = 4
    All     = Read | Write | Execute
};

// ═══════════════════════════════════════════════════════
// Operator overloading for bitwise operations
// ═══════════════════════════════════════════════════════
Permissions operator|(Permissions a, Permissions b) {
    return static_cast<Permissions>(
        static_cast<unsigned int>(a) | static_cast<unsigned int>(b)
    );
}

Permissions operator&(Permissions a, Permissions b) {
    return static_cast<Permissions>(
        static_cast<unsigned int>(a) & static_cast<unsigned int>(b)
    );
}

Permissions operator~(Permissions a) {
    return static_cast<Permissions>(~static_cast<unsigned int>(a));
}

bool hasPermission(Permissions current, Permissions check) {
    return (static_cast<unsigned int>(current) & 
            static_cast<unsigned int>(check)) != 0;
}

int main() {
    // Combine permissions
    Permissions userPerms = Permissions::Read | Permissions::Write;

    cout << "User permissions: " 
         << static_cast<unsigned int>(userPerms) << endl;  // 3

    // Check permissions
    if (hasPermission(userPerms, Permissions::Read)) {
        cout << "✓ User can read" << endl;
    }

    if (hasPermission(userPerms, Permissions::Write)) {
        cout << "✓ User can write" << endl;
    }

    if (!hasPermission(userPerms, Permissions::Execute)) {
        cout << "✗ User cannot execute" << endl;
    }

    // Add permission
    userPerms = userPerms | Permissions::Execute;
    
    if (hasPermission(userPerms, Permissions::Execute)) {
        cout << "✓ Execute permission added" << endl;
    }

    return 0;
}
```

``` cpp
    "Monday", "Tuesday", "Wednesday", "Thursday",
    "Friday", "Saturday", "Sunday"

// Helper to iterate
template<typename EnumType>
class EnumIterator {
private:
    int value;

public:
    EnumIterator(int v) : value(v) {}

    EnumType operator*() const {
        return static_cast<EnumType>(value);
    }

    EnumIterator& operator++() {
        ++value;
        return *this;
    }

    bool operator!=(const EnumIterator& other) const {
        return value != other.value;
    }
};

template<typename EnumType>
class EnumRange {
private:
    int start, end;

public:
    EnumRange(int s, int e) : start(s), end(e) {}

    EnumIterator<EnumType> begin() const {
        return EnumIterator<EnumType>(start);
    }

    EnumIterator<EnumType> end() const {
        return EnumIterator<EnumType>(end);
    }
};

int main() {

    // Method 1: Traditional loop

        DayOfWeek day = static_cast<DayOfWeek>(i);

    for (DayOfWeek day : days) {
```

``` cpp

#include <string>

// ═══════════════════════════════════════════════════════
// Traffic Light State Machine
// ═══════════════════════════════════════════════════════
enum class TrafficLightState {
    Red,
    Yellow,
    Green
};

class TrafficLight {
private:
    TrafficLightState currentState;

    string stateToString(TrafficLightState state) const {
        switch (state) {
            case TrafficLightState::Red:    return "🔴 RED";
            case TrafficLightState::Yellow: return "🟡 YELLOW";
            case TrafficLightState::Green:  return "🟢 GREEN";

        TrafficLightState nextState;

        switch (currentState) {
            case TrafficLightState::Red:
                nextState = TrafficLightState::Green;
                cout << "Red → Green (GO)" << endl;
                break;

            case TrafficLightState::Green:
                nextState = TrafficLightState::Yellow;
                cout << "Green → Yellow (CAUTION)" << endl;
                break;

            case TrafficLightState::Yellow:
                nextState = TrafficLightState::Red;
                cout << "Yellow → Red (STOP)" << endl;
                break;
        }

        currentState = nextState;
    }

    void display() const {
        cout << "Current state: " << stateToString(currentState) << endl;
    }

    TrafficLightState getState() const {
        return currentState;
    }

    bool canCross() const {
        return currentState == TrafficLightState::Green;
    }
};

// ═══════════════════════════════════════════════════════
// Order Processing State Machine
// ═══════════════════════════════════════════════════════

            case OrderState::PaymentConfirmed: return "Payment Confirmed";

            state = OrderState::PaymentPending;

            state = OrderState::PaymentConfirmed;

        return false;

            state = OrderState::Shipped;

        }
        return false;
    }

        if (state == OrderState::Shipped) {
            state = OrderState::Delivered;
            cout << "  → Order delivered" << endl;

        }
        return false;
    }

        if (state != OrderState::Delivered && 
            state != OrderState::Cancelled) {
            state = OrderState::Cancelled;
            cout << "  → Order cancelled" << endl;

        }
```

### 9. Best Practices
```cpp 
// ✅ DO: Use enum class for new code
enum class Status { Active, Inactive };

// ✅ DO: Specify underlying type when needed
enum class FileMode : uint8_t { Read, Write, Execute };

// ✅ DO: Use meaningful names
enum class UserRole { Admin, Editor, Viewer };  // Good
enum class UR { A, E, V };                      // Bad

// ✅ DO: Provide string conversion functions
string toString(UserRole role);
UserRole fromString(const string& str);

// ✅ DO: Use COUNT for iteration
enum class Color { Red, Green, Blue, COUNT };

// ❌ DON'T: Use plain enum in new code
// enum Color { RED, GREEN, BLUE };  // Avoid

// ❌ DON'T: Rely on implicit int conversion
// enum class Value { One = 1 };
// int x = Value::One;  // Error, use static_cast

// ✅ DO: Overload operators for bit flags
enum class Flags : unsigned int { A = 1, B = 2, C = 4 };
Flags operator|(Flags a, Flags b);

// ✅ DO: Use switch with enum class (compiler warns on missing cases)
void handle(Status status) {
    switch (status) {
        case Status::Active:   /*...*/ break;
        case Status::Inactive: /*...*/ break;
        // Compiler warns if case is missing
    }
}
```

### Summary:
| Use Case | Recommendation |
| :--- | :--- |
| **New code** | `enum class` (scoped, type-safe) |
| **Legacy code** | `enum` (for compatibility) |
| **Bit flags** | `enum class` with operator overloads |
| **API boundaries** | `enum class` with explicit conversions |
