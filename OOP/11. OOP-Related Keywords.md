## C++ Keywords - Comprehensive Guide
A complete reference covering all contexts: OOP, procedural, templates, memory management, and more.

### 1. this - Context Pointer
**In Classes (OOP):**
``` cpp
class Person {
    string name;
public:
    Person(string name) : name(name) {}  // Resolve naming conflict
    
    Person& setName(string name) {
        this->name = name;
        return *this;  // Enable method chaining
    }
    
    bool operator==(const Person& other) const {
        return this == &other;  // Check if same object
    }
};
```

**Lambda Captures (C++11):**
``` cpp
class Widget {
    int value = 10;
public:
    void foo() {
        auto lambda = [this]() {
            cout << value << endl;  // Captures 'this' pointer
        };
        lambda();
        
        // C++17: Capture *this by value
        auto lambda2 = [*this]() {
            cout << value << endl;  // Copy of object
        };
    }
};
```

### 2. new and delete - Memory Management
**Basic Allocation:**
``` cpp
// Single object
int* p = new int(42);
delete p;

// Array
int* arr = new int[10];
delete[] arr;

// No initialization
int* raw = new int;  // Uninitialized
delete raw;
```

**Placement New:**
``` cpp
#include <new>

alignas(int) char buffer[sizeof(int)];
int* p = new(buffer) int(42);  // Construct in existing memory
p->~int();  // Explicit destructor call (no delete)
```

**Overloading new/delete:**
``` cpp
class Custom {
public:
    void* operator new(size_t size) {
        cout << "Custom new: " << size << " bytes" << endl;
        return ::operator new(size);
    }
    
    void operator delete(void* ptr) {
        cout << "Custom delete" << endl;
        ::operator delete(ptr);
    }
};
```
**new vs malloc:**
``` cpp
// malloc: C-style, no constructors
int* p1 = (int*)malloc(sizeof(int));
free(p1);

// new: C++, calls constructors
int* p2 = new int(42);
delete p2;

class MyClass {
public:
    MyClass() { cout << "Constructor" << endl; }
};

// MyClass* obj1 = (MyClass*)malloc(sizeof(MyClass));  // ❌ No constructor
MyClass* obj2 = new MyClass();  // ✅ Constructor called

```

**nothrow new:**
``` cpp
int* p = new(nothrow) int[1000000000];  // Returns nullptr on failure
if (!p) {
    cout << "Allocation failed" << endl;
}
delete[] p;
```

### 3. default and delete
**Special Member Functions:**

``` cpp
class Widget {
public:
    Widget() = default;  // Compiler-generated
    ~Widget() = default;
    
    Widget(const Widget&) = delete;  // Disable copying
    Widget& operator=(const Widget&) = delete;
    
    Widget(Widget&&) = default;  // Enable moving
    Widget& operator=(Widget&&) = default;
};
```

**Delete Specific Overloads:**
``` cpp
void process(int x) { cout << "int: " << x << endl; }
void process(double x) = delete;  // Prevent implicit conversion

int main() {
    process(42);      // ✅ OK
    // process(3.14); // ❌ ERROR: Deleted function
}
```

**Delete Free Functions:**
``` cpp
void dangerousFunction() = delete;  // Cannot be called at all

template<typename T>
void func(T value) { }

template<>
void func<bool>(bool) = delete;  // Disable for bool
```

### 4. final
**Prevent Inheritance:**
``` cpp
class Base final {
    virtual void foo() {}
};

// class Derived : public Base { };  // ❌ ERROR

```

**Prevent Overriding:**
``` cpp
class Base {
    virtual void canOverride() {}
    virtual void cannotOverride() final {}
};

class Derived : public Base {
    void canOverride() override {}
    // void cannotOverride() override {}  // ❌ ERROR
};
```

**Performance Optimization:**
``` cpp
class FastClass {
    virtual void normalVirtual() {}
    virtual void fastFinal() final {
        // Compiler can devirtualize and inline
    }
};
```

### 5. static
**Static Variables in Functions:**
``` cpp
void counter() {
    static int count = 0;  // Initialized once
    cout << ++count << endl;
}

int main() {
    counter();  // 1
    counter();  // 2
    counter();  // 3
}
```

**Static Class Members:**
``` cpp
class Config {
public:
    static int instanceCount;
    static const double PI;  // Const static can be initialized in class (C++11)
    
    static void printCount() {
        cout << instanceCount << endl;
    }
};

int Config::instanceCount = 0;
const double Config::PI = 3.14159;
```

**Static Local Variables (Thread-Safe Initialization C++11):**
``` cpp
class Singleton {
public:
    static Singleton& getInstance() {
        static Singleton instance;  // Thread-safe in C++11
        return instance;
    }
private:
    Singleton() {}
};
```

**Static in File Scope (Internal Linkage):**

``` cpp
// file1.cpp
static int helper = 42;  // Only visible in this file
static void utilFunc() {}  // Internal linkage

// file2.cpp
static int helper = 100;  // Different variable, same name OK
```

**Static Assert (Compile-Time):**
``` cpp
static_assert(sizeof(int) == 4, "int must be 4 bytes");
static_assert(sizeof(void*) == 8, "64-bit pointers required");

template<typename T>
void foo(T value) {
    static_assert(std::is_integral<T>::value, "T must be integral");
}
```

### 6. mutable
**In Classes:**
``` cpp
class Logger {
    mutable mutex mtx;
    mutable int logCount;
    string message;
    
public:
    void log(const string& msg) const {
        lock_guard<mutex> lock(mtx);  // Lock mutable mutex
        ++logCount;  // Modify mutable member
        // message = msg;  // ❌ ERROR: Can't modify non-mutable
    }
};

```

**Lambda Captures (C++11):**
``` cpp
int x = 10;
auto lambda = [x]() mutable {
    x++;  // Modifies lambda's copy of x
    cout << x << endl;
};

lambda();  // 11
cout << x << endl;  // Still 10 (original unchanged)
```

### 7. const
**Variables:**
``` cpp
const int MAX = 100;
// MAX = 200;  // ❌ ERROR

int value = 50;
const int* ptr1 = &value;  // Pointer to const int
// *ptr1 = 60;  // ❌ ERROR
ptr1 = nullptr;  // ✅ OK

int* const ptr2 = &value;  // Const pointer to int
*ptr2 = 60;  // ✅ OK
// ptr2 = nullptr;  // ❌ ERROR

const int* const ptr3 = &value;  // Const pointer to const int
// *ptr3 = 70;  // ❌ ERROR
// ptr3 = nullptr;  // ❌ ERROR
```

**Functions:**
``` cpp
class Array {
    int data[10];
public:
    // Const member function
    int get(int index) const {
        return data[index];
    }
    
    // Non-const member function
    void set(int index, int value) {
        data[index] = value;
    }
    
    // Const overload
    const int& operator[](int i) const { return data[i]; }
    int& operator[](int i) { return data[i]; }
};

// Const parameters
void print(const string& str) {  // Can't modify str
    cout << str << endl;
}

// Const return value
const int& getMax(const int& a, const int& b) {
    return (a > b) ? a : b;
}
```

**constexpr (C++11):**
``` cpp
constexpr int square(int x) {
    return x * x;
}

constexpr int SIZE = square(10);  // Evaluated at compile time
int arr[SIZE];  // OK: SIZE is compile-time constant

// constexpr constructor (C++11)
class Point {
    int x_, y_;
public:
    constexpr Point(int x, int y) : x_(x), y_(y) {}
    constexpr int x() const { return x_; }
    constexpr int y() const { return y_; }
};

constexpr Point origin(0, 0);
```

**const_cast:**
``` cpp
void legacyFunc(char* str);  // Can't change signature

void modernFunc(const char* str) {
    legacyFunc(const_cast<char*>(str));  // Remove const
}
```

### 8. explicit
**Constructors:**
``` cpp
class String {
public:
    explicit String(int size) { }  // Prevent implicit conversion
    String(const char* str) { }    // Allow implicit conversion
};

void process(String s) { }

int main() {
    process("Hello");  // ✅ OK: Implicit conversion from const char*
    // process(10);    // ❌ ERROR: Explicit constructor
    process(String(10));  // ✅ OK: Explicit conversion
}
```

**Conversion Operators:**
``` cpp
class Fraction {
    int numerator, denominator;
public:
    explicit operator double() const {
        return static_cast<double>(numerator) / denominator;
    }
};

Fraction f(3, 4);
// double d = f;  // ❌ ERROR: Explicit conversion
double d = static_cast<double>(f);  // ✅ OK
```

### 9. inline
**Functions:**
``` cpp
inline int add(int a, int b) {
    return a + b;  // Hint to compiler to inline
}

// Member functions defined in class are implicitly inline
class Math {
public:
    int multiply(int a, int b) {  // Implicitly inline
        return a * b;
    }
};
```

**Variables (C++17):**
``` cpp
// header.h
inline int globalVar = 42;  // OK in C++17, no ODR violation

class Config {
public:
    static inline int maxSize = 1000;  // C++17
    inline static const double PI = 3.14159;  // C++17
};
```

**Inline Namespaces (C++11):**
``` cpp
namespace MyLib {
    inline namespace v2 {
        void feature() { cout << "v2" << endl; }
    }
    namespace v1 {
        void feature() { cout << "v1" << endl; }
    }
}

MyLib::feature();     // Calls v2::feature (inline namespace)
MyLib::v1::feature();  // Calls v1::feature (explicit)
```

### 10. template and typename
**Function Templates:**
``` cpp
template<typename T>
T max(T a, T b) {
    return (a > b) ? a : b;
}

// Multiple parameters
template<typename T, typename U>
auto add(T a, U b) -> decltype(a + b) {
    return a + b;
}

// Non-type parameters
template<typename T, int N>
class Array {
    T data[N];
};
```

**Class Templates:**
``` cpp
template<typename T>
class Stack {
    vector<T> elements;
public:
    void push(T elem) { elements.push_back(elem); }
    T pop() { 
        T elem = elements.back();
        elements.pop_back();
        return elem;
    }
};
```

**typename for Dependent Types:**
``` cpp
template<typename Container>
void print(const Container& c) {
    typename Container::const_iterator it;  // 'typename' required
    for (it = c.begin(); it != c.end(); ++it) {
        cout << *it << " ";
    }
}
```

**Template Specialization:**
``` cpp
// Primary template
template<typename T>
class Printer {
public:
    void print(T value) { cout << value << endl; }
};

// Full specialization
template<>
class Printer<bool> {
public:
    void print(bool value) { 
        cout << (value ? "true" : "false") << endl; 
    }
};

// Partial specialization
template<typename T>
class Printer<T*> {
public:
    void print(T* ptr) { cout << "Pointer: " << ptr << endl; }
};
```

**Variadic Templates (C++11):**
``` cpp
template<typename... Args>
void print(Args... args) {
    (cout << ... << args) << endl;  // C++17 fold expression
}

print(1, 2.5, "Hello", 'A');
```

### 11. using and typedef
**Type Aliases:**
``` cpp
// typedef (old style)
typedef unsigned long ulong;
typedef vector<int> IntVec;

// using (C++11, preferred)
using ulong = unsigned long;
using IntVec = vector<int>;
```

**Template Aliases (using only):**
``` cpp
template<typename T>
using Vec = vector<T>;

template<typename T>
using Ptr = unique_ptr<T>;

Vec<int> numbers;
Ptr<Widget> widget = make_unique<Widget>();
```

**Using Declarations:**
``` cpp
using std::cout;
using std::endl;
using std::vector;

cout << "Hello" << endl;
vector<int> v;
```

**Using Directives:**
``` cpp
using namespace std;  // Brings entire namespace

namespace MyLib {
    void func() {}
}

using namespace MyLib;
func();  // No need for MyLib::
```

**Type Aliases in Classes:**
``` cpp
template<typename T>
class Container {
public:
    using value_type = T;
    using size_type = size_t;
    using iterator = T*;
};

Container<int>::value_type x = 42;
```

### 12. namespace
**Basic Namespaces:**
``` cpp
namespace Math {
    const double PI = 3.14159;
    
    double square(double x) {
        return x * x;
    }
    
    class Calculator {
    public:
        static int add(int a, int b) { return a + b; }
    };
}

// Usage
double area = Math::PI * Math::square(5);
int sum = Math::Calculator::add(10, 20);
```

**Nested Namespaces:**
``` cpp
namespace Company {
    namespace Project {
        namespace Module {
            void function() { }
        }
    }
}

// C++17: Nested namespace definition
namespace Company::Project::Module {
    void anotherFunction() { }
}
```

**Anonymous Namespaces (Internal Linkage):**

``` cpp
// file.cpp
namespace {
    int helper() { return 42; }  // Only visible in this file
    class InternalClass { };     // Internal linkage
}

void publicFunc() {
    helper();  // OK within same file
}
```

**Namespace Aliases:**
``` cpp
namespace VeryLongNamespaceName {
    void feature() { }
}

namespace Short = VeryLongNamespaceName;
Short::feature();
```

**Inline Namespaces (Versioning):**
``` cpp
namespace MyLib {
    inline namespace v2 {
        void api() { cout << "v2 API" << endl; }
    }
    
    namespace v1 {
        void api() { cout << "v1 API" << endl; }
    }
}

MyLib::api();     // Calls v2::api (default)
MyLib::v1::api();  // Explicitly call v1
```

**ADL (Argument-Dependent Lookup):**
``` cpp
namespace NS {
    struct Type { };
    void func(Type) { cout << "NS::func" << endl; }
}

int main() {
    NS::Type obj;
    func(obj);  // ADL finds NS::func without qualification
}
```

### 13. noexcept
**Basic Usage:**
``` cpp
void safe() noexcept {
    // Guaranteed not to throw
}

void unsafe() {
    throw runtime_error("Error");
}

void conditional() noexcept(sizeof(int) == 4) {
    // noexcept if condition is true
}
```

**Querying noexcept:**
``` cpp
void func() noexcept { }

static_assert(noexcept(func()), "Should be noexcept");
static_assert(!noexcept(throw 1), "Throwing is not noexcept");
```

**Move Operations:**
``` cpp
class Widget {
public:
    Widget(Widget&& other) noexcept {
        // Move constructor should be noexcept
    }
    
    Widget& operator=(Widget&& other) noexcept {
        // Move assignment should be noexcept
        return *this;
    }
};

// Affects std::vector performance
vector<Widget> vec;
vec.push_back(Widget());  // Uses move if noexcept, else copy

```


**Conditional noexcept:**
``` cpp
template<typename T>
class Container {
public:
    void swap(Container& other) 
        noexcept(noexcept(std::swap(data, other.data))) {
        std::swap(data, other.data);
    }
private:
    T data;
};
```

**Destructors (implicitly noexcept):**
``` cpp
class Resource {
public:
    ~Resource() noexcept {  // Implicitly noexcept
        // Cleanup code
    }
    
    // ❌ Don't throw in destructors!
    // ~Resource() {
    //     throw runtime_error("Bad");  // Dangerous!
    // }
};
```

### 14. virtual - Runtime Polymorphism
``` Cpp

class Base {
public:
    virtual void foo() { cout << "Base" << endl; }
    virtual void bar() = 0;  // Pure virtual
    virtual ~Base() = default;  // Virtual destructor
};

class Derived : public Base {
public:
    void foo() override { cout << "Derived" << endl; }
    void bar() override { cout << "Bar" << endl; }
};
```

### 15. override - Explicit Override (C++11)
``` cpp

class Base {
    virtual void func(int) { }
};

class Derived : public Base {
    void func(int) override { }  // ✅ Explicit override
    // void func(double) override { }  // ❌ ERROR: Not overriding
};
```

### 16. friend - Grant Access
``` cpp

class Box {
private:
    int value;
    
    friend class BoxInspector;
    friend void print(const Box&);
    
public:
    Box(int v) : value(v) { }
};

class BoxInspector {
public:
    void inspect(const Box& b) {
        cout << b.value << endl;  // Access private member
    }
};

void print(const Box& b) {
    cout << "Value: " << b.value << endl;  // Friend function
}
```

### 17. auto - Type Deduction (C++11)
``` cpp

auto x = 42;              // int
auto y = 3.14;            // double
auto s = "Hello";         // const char*
auto v = vector<int>{};   // vector<int>

// With functions
auto add(int a, int b) -> int {
    return a + b;
}

// Generic lambdas (C++14)
auto lambda = [](auto x, auto y) { return x + y; };
```

### 18. decltype - Type of Expression (C++11)
``` cpp

int x = 10;
decltype(x) y = 20;  // y is int

template<typename T, typename U>
auto add(T a, U b) -> decltype(a + b) {
    return a + b;
}

// decltype(auto) (C++14)
decltype(auto) func() {
    int x = 10;
    return x;  // Returns int, not int&
}
```

### 19. constexpr - Compile-Time Evaluation (C++11)
``` cpp

constexpr int factorial(int n) {
    return (n <= 1) ? 1 : n * factorial(n - 1);
}

constexpr int val = factorial(5);  // Computed at compile time
int arr[val];  // OK: val is compile-time constant
```

### 20. nullptr - Null Pointer (C++11)
```cpp

int* p1 = NULL;      // Old style (actually 0)
int* p2 = nullptr;   // C++11

void func(int) { cout << "int" << endl; }
void func(int*) { cout << "pointer" << endl; }

func(NULL);     // Ambiguous! Calls func(int)
func(nullptr);  // ✅ Calls func(int*)
```

### 21. volatile - Prevent Optimization
``` cpp

volatile int flag = 0;  // Hardware register, don't optimize

void wait() {
    while (flag == 0) {
        // Without volatile, the compiler might optimize away the loop
    }
}

// Interrupt handler
void interrupt_handler() {
    flag = 1;  // Modified by external event
}
```
